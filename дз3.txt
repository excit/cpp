// 1) Вводится число. Вывести сумму всех чисел в диапазоне от единицы до введённого числа. 

void number1()
{
	printf("Введите число для подсчёта суммы в диапозоне от 1: ");

	int b, i;
	scanf_s("%d", &b);
		
		int sum = 0; 

		for (i = b; i > 0; i--) {

			sum += i;
		}
		printf("Сумма чисел в заданном диапозоне равна %d \n", sum);
	
	system("pause");
}

// 2) Вводится число. Вывести количество тех чисел в диапазоне от единицы до введённого числа, которые делятся на 3, но не делятся на 5.

void number2()
{
	int a, i, q;
	q = 0;

	printf("Введите число в диапозоне от единицы: ");
	scanf_s("%d", &a);

	for (i = a; i > 0; i--){

		if ((i % 3 == 0) && (i % 5 != 0)) {

			q += 1;
		}
	}

	printf("Количество чисел, делящихся на 5 и не кратных 3 равно %d \n", q);
		system("pause");
	
}

// 3) Вводится число. Вывести произведение тех чисел в диапазоне от единицы до введённого числа, которые не делятся ни на 2, ни на 3.

void number3()
{
	int a, i, pr;
	pr = 1;

	printf("Введите число в диапозоне от единицы: ");
	scanf_s("%d", &a);

	for (i = a; i > 0; i--){

		if ((i % 3 != 0) && (i % 2 != 0)) {

			pr = pr * i;
		}
	}
	printf("Произведение чисел, которые не делятся ни на 2 ни на 3 = %d \n", pr);

	system("pause");
}

// 4) Вводится число. Вывести, сколько у него делителей.

void number4()
{
	int a, i, k;
	k = 0; 

	printf("Введите число: ");
	scanf_s("%d", &a);

	for (i = a; i >= 1; i--) 
	{
		if (a % i == 0) {

			k++;
		}
	}
	printf("Количество делителей равно %d \n", k);

	system("pause");
}

// 5) Вводится число. Вывести сумму тех его делителей, которые оканчиваются на цифру 3.

void number5()

{
	int a, i, s;
	s = 0;

	printf("Введите число: ");
	scanf_s("%d", &a);

	for (i = a; i >= 1; i--)

	{
		if ((a % i == 0) && (i % 10 == 3)) 
		{
			s = s + i;
		}
	}

	printf("Сумма делителей числа, оканчивающихся на 3 равна %d \n", s);

	system("pause");
}
